1: package Gimble::Page::ViewSrc::Base;

  3: use base qw(Gimble::Dispatch Gimble::Page::Base);


  6: sub find_module {
  7:   my $module_name = shift;
  8:   $module_name =~ s!::!/!g;
  9:   $module_name = "$module_name.pm";
 10:   $INC{$module_name};
 11: }

 13: sub template {

 15:   my $self = shift;

 17:   my $module = $self->dispatch(  $self->param('otask')  ) ;

 19:   my $file = find_module $module;

 21:   my $html = `perl2html -c -n $file`;

 23:   use HTML::TreeBuilder;

 25:   HTML::TreeBuilder->new_from_content($html);

 27: }

 29: 1;
